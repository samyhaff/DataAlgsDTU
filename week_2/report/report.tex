\textbf{5.2}

$$
W[i,j] =
\begin{cases}
0 & \text{for } $(i, j) = (1, 1)$ \\
W[i,j-1] + R[i,j-1] & \text{for } i = 1 \\
W[i-1,j] + D[i-1,j] & \text{for } j = 1 \\
\max(W[i-1, j] + D[i-1,j], W[i,j-1] + R[i,j-1]) & \text{otherwise}
\end{cases}
$$
